        """禁用插件时调用"""

### 2.5 扩展功能详细设计

#### 功能 1: 配置备份/恢复系统

```python
# features/backup.py
class BackupManager:
    """配置备份管理器"""
    
    def create_backup(self, name: str, include_logs: bool = False) -> str:
        """创建备份"""
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        backup_file = f"openclaw_backup_{name}_{timestamp}.zip"
        
        with zipfile.ZipFile(backup_file, 'w') as zf:
            # 备份配置
            zf.write(CONFIG_PATH, 'openclaw.json')
            # 备份技能
            zf.write(SKILLS_DIR, 'skills/')
            # 可选：备份日志
            if include_logs:
                zf.write(LOGS_DIR, 'logs/')
        
        return backup_file
    
    def restore_backup(self, backup_file: str) -> bool:
        """恢复备份"""
        with zipfile.ZipFile(backup_file, 'r') as zf:
            # 验证备份完整性
            if not self._validate_backup(zf):
                return False
            
            # 恢复前创建当前配置的备份
            self.create_backup('auto_before_restore')
            
            # 解压恢复
            zf.extractall(CONFIG_DIR)
        
        return True
```

#### 功能 2: 多配置环境

```python
# features/profiles.py
class ProfileManager:
    """配置环境管理器"""
    
    PROFILES_DIR = os.path.expanduser('~/.openclaw/profiles')
    
    def create_profile(self, name: str, base_profile: Optional[str] = None):
        """创建新环境"""
        profile_path = os.path.join(self.PROFILES_DIR, name)
        os.makedirs(profile_path, exist_ok=True)
        
        if base_profile:
            # 复制基础配置
            shutil.copytree(
                os.path.join(self.PROFILES_DIR, base_profile),
                profile_path
            )
        else:
            # 创建默认配置
            self._create_default_config(profile_path)
    
    def switch_profile(self, name: str):
        """切换环境"""
        # 保存当前配置
        self._save_current_profile()
        
        # 加载新配置
        profile_path = os.path.join(self.PROFILES_DIR, name)
        self._load_profile(profile_path)
        
        # 重启服务
        self._restart_services()
```

#### 功能 3: 云同步系统

```python
# features/cloud_sync.py
class CloudSyncManager:
    """云同步管理器"""
    
    def __init__(self, auth_token: str):
        self.client = CloudAPIClient(token=auth_token)
    
    def sync_up(self):
        """上传到云端"""
        # 打包配置
        config_package = self._pack_config()
        
        # 上传到云端
        self.client.upload_config(config_package)
        
        # 更新同步时间
        self._update_sync_timestamp()
    
    def sync_down(self):
        """从云端下载"""
        # 获取云端配置
        remote_config = self.client.download_config()
        
        # 合并配置（本地优先或云端优先）
        merged_config = self._merge_configs(local, remote_config)
        
        # 应用配置
        self._apply_config(merged_config)
    
    def auto_sync(self, interval_minutes: int = 30):
        """自动同步"""
        def sync_job():
            if self._has_local_changes():
                self.sync_up()
            if self._has_remote_changes():
                self.sync_down()
        
        schedule.every(interval_minutes).minutes.do(sync_job)
```

#### 功能 4: 智能诊断与修复

```python
# features/smart_diagnose.py
class SmartDiagnoseEngine:
    """智能诊断引擎"""
    
    RULES = [
        {
            'id': 'node_version',
            'check': lambda: check_node_version(),
            'fix': lambda: install_node_latest(),
            'severity': 'high',
        },
        {
            'id': 'config_permission',
            'check': lambda: check_config_permission(),
            'fix': lambda: fix_config_permission(),
            'severity': 'medium',
        },
        # ... 更多规则
    ]
    
    def diagnose(self) -> List[Issue]:
        """执行诊断"""
        issues = []
        for rule in self.RULES:
            if not rule['check']():
                issues.append(Issue(
                    id=rule['id'],
                    severity=rule['severity'],
                    fix_available=bool(rule['fix'])
                ))
        return issues
    
    def auto_fix(self, issue_ids: List[str]):
        """自动修复问题"""
        for issue_id in issue_ids:
            rule = self._get_rule(issue_id)
            if rule and rule['fix']:
                try:
                    rule['fix']()
                    self._log_fix_success(issue_id)
                except Exception as e:
                    self._log_fix_failure(issue_id, e)
```
