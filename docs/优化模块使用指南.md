# OpenClaw Butler - 优化模块使用指南

## 概述

本优化方案包含以下核心模块：

### 1. 诊断工具优化
- `diagnose/health_score.py` - 健康度评分系统
- `diagnose/smart_fix.py` - 智能修复助手
- `ui/health_dashboard.py` - 健康度仪表盘UI

### 2. 技能安装优化
- `skills/install_queue.py` - 安装队列管理器
- `skills/dependency_checker.py` - 依赖检查器
- `skills/error_handler.py` - 错误处理器
- `ui/skill_market.py` - 技能市场UI

## 快速开始

### 1. 健康度评分系统

```python
from diagnose.health_score import HealthScoreCalculator, HealthScoreHistory

# 计算健康度评分
calculator = HealthScoreCalculator()
results = [
    {"name": "Node.js", "status": "pass", "category": "environment", "weight": 10},
    {"name": "OpenClaw CLI", "status": "pass", "category": "tools", "weight": 10},
    {"name": "Config", "status": "warn", "category": "configuration", "weight": 5, "message": "发现未知键"},
]

score_result = calculator.calculate(results)
print(f"健康度评分: {score_result.total_score}分")
print(f"等级: {score_result.grade.grade} - {score_result.grade.label}")

# 保存历史记录
history = HealthScoreHistory()
history.save(score_result)

# 获取趋势
trend = history.get_trend(days=30)
```

### 2. 智能修复助手

```python
from diagnose.smart_fix import SmartFixAssistant

# 分析问题并提供修复方案
assistant = SmartFixAssistant()

issue = {
    "name": "Config Error",
    "status": "fail",
    "message": "models 包含未识别键: default, thinking",
    "category": "configuration"
}

result = assistant.analyze_and_fix(issue)
print(f"修复结果: {result.message}")
print(f"修复级别: {result.level.value}")

# 生成修复脚本
issues = [issue]
script = assistant.generate_fix_script(issues, platform="windows")
print(script)
```

### 4. 依赖检查器

```python
from skills.dependency_checker import DependencyChecker, check_skill_dependencies

# 检查技能依赖
checker = DependencyChecker()
report = checker.check_before_install("playwright-mcp")

print(f"是否可以安装: {report.can_install}")
print(f"检查项数: {len(report.checks)}")
print(f"可自动修复: {len(report.auto_fixable)}")

# 自动修复依赖
if report.auto_fixable:
    success, errors = checker.auto_fix_dependencies(report)
    if success:
        print("依赖修复成功")
    else:
        print(f"修复失败: {errors}")

# 便捷函数
report = check_skill_dependencies("skill-vetter")
if report.can_install:
    print("可以安装")
```

## 集成到 OpenClaw Butler

### 1. 修改 DiagnosePanel

在 `openclaw_assistant.py` 中，替换现有的 `DiagnosePanel`：

```python
# 导入新模块
from diagnose.health_score import HealthScoreCalculator, HealthScoreHistory
from diagnose.smart_fix import SmartFixAssistant
from ui.health_dashboard import HealthDashboard

class DiagnosePanel(BasePanel):
    def __init__(self, parent, app, **kwargs):
        super().__init__(parent, app, **kwargs)
        
        # 使用新的健康度仪表盘
        self.dashboard = HealthDashboard(
            self,
            on_diagnose=self._run_diagnose,
            on_fix=self._auto_fix
        )
        self.dashboard.pack(fill=tk.BOTH, expand=True)
        
        # 初始化组件
        self.calculator = HealthScoreCalculator()
        self.history = HealthScoreHistory()
        self.fix_assistant = SmartFixAssistant()
    
    def _run_diagnose(self):
        """运行诊断"""
        # 运行原有诊断逻辑
        results = self._get_diagnose_results()
        
        # 计算健康度评分
        score_result = self.calculator.calculate(results)
        
        # 更新UI
        self.dashboard.update_score(
            score_result.total_score,
            score_result.grade.grade,
            score_result.grade.label,
            score_result.grade.color
        )
        
        # 更新分类
        categories = [
            {"name": c.name, "score": c.score}
            for c in score_result.categories
        ]
        self.dashboard.update_categories(categories)
        
        # 更新问题列表
        issues = []
        for cat in score_result.categories:
            issues.extend(cat.issues)
        self.dashboard.update_issues(issues, on_view=self._view_issue, on_fix_single=self._fix_single_issue)
        
        # 保存历史
        self.history.save(score_result)
    
    def _auto_fix(self):
        """一键修复"""
        # 获取当前问题
        issues = self._get_current_issues()
        
        for issue in issues:
            result = self.fix_assistant.analyze_and_fix(issue)
            
            if result.level.value == "auto":
                # 自动修复成功
                print(f"已自动修复: {result.message}")
            elif result.level.value == "guided":
                # 需要用户确认
                if messagebox.askyesno("修复确认", result.message):
                    # 执行修复
                    pass
            else:
                # 手动修复
                messagebox.showinfo("手动修复", result.details)
```

### 2. 修改技能管理面板

```python
# 导入新模块
from skills.install_queue import InstallQueueManager
from skills.dependency_checker import DependencyChecker
from skills.error_handler import InstallErrorHandler, OfflineInstaller
from ui.skill_market import SkillMarket

class SkillsPanel(BasePanel):
    def __init__(self, parent, app, **kwargs):
        super().__init__(parent, app, **kwargs)
        
        # 使用新的技能市场UI
        self.market = SkillMarket(self)
        self.market.pack(fill=tk.BOTH, expand=True)
        
        # 初始化组件
        self.queue_manager = InstallQueueManager()
        self.dependency_checker = DependencyChecker()
        self.error_handler = InstallErrorHandler()
        self.offline_installer = OfflineInstaller()
        
        # 设置队列回调
        self.queue_manager.on_task_complete = self._on_install_complete
        self.queue_manager.on_task_failed = self._on_install_failed
        self.queue_manager.on_queue_update = self._on_queue_update
        
        self.queue_manager.start()
    
    def install_skill(self, skill_info: dict):
        """安装技能"""
        skill_id = skill_info["id"]
        
        # 1. 检查依赖
        report = self.dependency_checker.check_before_install(skill_id)
        
        if not report.can_install:
            # 显示依赖检查失败
            messagebox.showerror(
                "依赖检查失败",
                f"无法安装 {skill_id}:\n{report.summary}"
            )
            return
        
        # 2. 自动修复依赖
        if report.auto_fixable:
            success, errors = self.dependency_checker.auto_fix_dependencies(report)
            if not success:
                messagebox.showwarning(
                    "依赖修复失败",
                    f"部分依赖无法自动修复:\n{errors}"
                )
        
        # 3. 添加到安装队列
        task = self.queue_manager.add_task(
            skill_id=skill_id,
            skill_name=skill_info.get("name", skill_id),
            priority=5
        )
        
        # 4. 更新UI
        self.market.queue_panel.add_task(task.id, task.skill_name)
    
    def _on_install_complete(self, task):
        """安装完成回调"""
        self.market.queue_panel.update_task(
            task.id, "completed", progress=100
        )
        messagebox.showinfo("安装成功", f"技能 {task.skill_name} 安装完成")
    
    def _on_install_failed(self, task, error_msg):
        """安装失败回调"""
        # 分析错误
        analysis = self.error_handler.analyze_error(error_msg, task.skill_id)
        
        self.market.queue_panel.update_task(
            task.id, "failed", message=analysis.message
        )
        
        # 显示错误信息和解决方案
        solutions = "\n".join(f"• {s}" for s in analysis.solutions)
        messagebox.showerror(
            "安装失败",
            f"技能 {task.skill_name} 安装失败\n\n"
            f"错误: {analysis.message}\n\n"
            f"建议解决方案:\n{solutions}"
        )
    
    def _on_queue_update(self, stats):
        """队列更新回调"""
        percentage = int(stats.overall_progress * 100)
        self.market.queue_panel.update_overall_progress(percentage)
```
