# OpenClaw Hub - 腾讯云云开发实现方案

## 一、为什么选择腾讯云云开发

### 1.1 云开发优势

| 特性 | 说明 | 适合场景 |
|------|------|---------|
| **免运维** | 无需管理服务器 | 小团队/快速启动 |
| **按量付费** | 用多少付多少 | 成本控制 |
| **一体化** | 数据库+存储+函数+托管 | 快速开发 |
| **自动扩缩容** | 根据流量自动调整 | 流量波动大 |
| **微信生态** | 天然支持微信登录 | 国内用户 |

### 1.2 成本对比

| 方案 | 月费用（预估） | 适合阶段 |
|------|---------------|---------|
| 自建服务器 | ¥5,000+ | 大规模生产 |
| 云服务器 | ¥2,000+ | 中小规模 |
| **云开发** | **¥200-800** | **初创/MVP** |

### 3.2 云函数实现

#### 用户服务 (functions/user-api/index.js)

```javascript
const cloud = require('wx-server-sdk');
const crypto = require('crypto');
const jwt = require('jsonwebtoken');

// 初始化云开发
cloud.init({
  env: cloud.DYNAMIC_CURRENT_ENV
});

const db = cloud.database();
const _ = db.command;

// JWT 密钥
const JWT_SECRET = process.env.JWT_SECRET || 'your-secret-key';

// 主入口
exports.main = async (event, context) => {
  const { action, data } = event;
  
  try {
    switch (action) {
      case 'register':
        return await register(data);
      case 'login':
        return await login(data);
      case 'getUserInfo':
        return await getUserInfo(data);
      case 'updateUserInfo':
        return await updateUserInfo(data);
      case 'wxLogin':
        return await wxLogin(data);
      default:
        return { code: 404, message: '未知操作' };
    }
  } catch (error) {
    console.error('Error:', error);
    return { code: 500, message: '服务器错误', error: error.message };
  }
};

// 用户注册
async function register(data) {
  const { username, email, password } = data;
  
  // 检查用户是否存在
  const existUser = await db.collection('users')
    .where(_.or([
      { username },
      { email }
    ]))
    .get();
  
  if (existUser.data.length > 0) {
    return { code: 400, message: '用户名或邮箱已存在' };
  }
  
  // 密码加密
  const passwordHash = crypto
    .createHash('sha256')
    .update(password + 'salt')
    .digest('hex');
  
  // 创建用户
  const result = await db.collection('users').add({
    data: {
      username,
      email,
      passwordHash,
      avatar: '',
      role: 'user',
      membershipLevel: 'free',
      membershipExpireAt: null,
      createdAt: db.serverDate(),
      updatedAt: db.serverDate()
    }
  });
  
  // 生成 Token
  const token = jwt.sign(
    { userId: result._id, username },
    JWT_SECRET,
    { expiresIn: '7d' }
  );
  
  return {
    code: 200,
    message: '注册成功',
    data: {
      userId: result._id,
      username,
      token
    }
  };
}

// 用户登录
async function login(data) {
  const { email, password } = data;
  
  // 密码加密
  const passwordHash = crypto
    .createHash('sha256')
    .update(password + 'salt')
    .digest('hex');
  
  // 查询用户
  const user = await db.collection('users')
    .where({
      email,
      passwordHash
    })
    .get();
  
  if (user.data.length === 0) {
    return { code: 401, message: '邮箱或密码错误' };
  }
  
  const userInfo = user.data[0];
  
  // 生成 Token
  const token = jwt.sign(
    { userId: userInfo._id, username: userInfo.username },
    JWT_SECRET,
    { expiresIn: '7d' }
  );
  
  return {
    code: 200,
    message: '登录成功',
    data: {
      userId: userInfo._id,
      username: userInfo.username,
      email: userInfo.email,
      avatar: userInfo.avatar,
      membershipLevel: userInfo.membershipLevel,
      token
    }
  };
}

// 微信登录
async function wxLogin(data) {
  const { code } = data;
  
  // 调用微信接口获取 openid
  const wxRes = await cloud.callFunction({
    name: 'wx-auth',
    data: { code }
  });
  
  const { openid, unionid } = wxRes.result;
  
  // 查找或创建用户
  let user = await db.collection('users')
    .where({ wxOpenId: openid })
    .get();
  
  if (user.data.length === 0) {
    // 创建新用户
    const result = await db.collection('users').add({
      data: {
        username: `wx_${openid.substr(-8)}`,
        wxOpenId: openid,
        wxUnionId: unionid,
        avatar: '',
        role: 'user',
        membershipLevel: 'free',
        createdAt: db.serverDate(),
        updatedAt: db.serverDate()
      }
    });
    
    user = { data: [{ _id: result._id, username: `wx_${openid.substr(-8)}` }] };
  }
  
  const userInfo = user.data[0];
  
  // 生成 Token
  const token = jwt.sign(
    { userId: userInfo._id, username: userInfo.username },
    JWT_SECRET,
    { expiresIn: '7d' }
  );
  
  return {
    code: 200,
    message: '登录成功',
    data: {
      userId: userInfo._id,
      username: userInfo.username,
      token
    }
  };
}

// 获取用户信息
async function getUserInfo(data) {
  const { userId } = data;
  
  const user = await db.collection('users').doc(userId).get();
  
  if (!user.data) {
    return { code: 404, message: '用户不存在' };
  }
  
  // 不返回敏感信息
  const { passwordHash, ...userInfo } = user.data;
  
  return {
    code: 200,
    data: userInfo
  };
}

// 更新用户信息
async function updateUserInfo(data) {
  const { userId, updateData } = data;
  
  await db.collection('users').doc(userId).update({
    data: {
      ...updateData,
      updatedAt: db.serverDate()
    }
  });
  
  return { code: 200, message: '更新成功' };
}
```

#### 技能服务 (functions/skill-api/index.js)

```javascript
const cloud = require('wx-server-sdk');

cloud.init({
  env: cloud.DYNAMIC_CURRENT_ENV
});

const db = cloud.database();
const _ = db.command;

exports.main = async (event, context) => {
  const { action, data } = event;
  
  try {
    switch (action) {
      case 'getSkills':
        return await getSkills(data);
      case 'getSkillDetail':
        return await getSkillDetail(data);
      case 'searchSkills':
        return await searchSkills(data);
      case 'createSkill':
        return await createSkill(data);
      case 'updateSkill':
        return await updateSkill(data);
      case 'getPopularSkills':
        return await getPopularSkills(data);
      case 'getNewSkills':
        return await getNewSkills(data);
      default:
        return { code: 404, message: '未知操作' };
    }
  } catch (error) {
    console.error('Error:', error);
    return { code: 500, message: '服务器错误' };
  }
};

// 获取技能列表
async function getSkills(data) {
  const { page = 1, size = 20, category, sort = 'downloads' } = data;
  
  let where = { status: 'approved' };
  if (category && category !== 'all') {
    where.category = category;
  }
  
  const skip = (page - 1) * size;
  
  const skills = await db.collection('skills')
    .where(where)
    .orderBy(sort, 'desc')
    .skip(skip)
    .limit(size)
    .get();
  
  const total = await db.collection('skills')
    .where(where)
    .count();
  
  return {
    code: 200,
    data: skills.data,
    pagination: {
      page,
      size,
      total: total.total
    }
  };
}

// 获取技能详情
async function getSkillDetail(data) {
  const { skillId } = data;
  
  const skill = await db.collection('skills')
    .where({ skillId })
    .get();
  
  if (skill.data.length === 0) {
    return { code: 404, message: '技能不存在' };
  }
  
  // 增加浏览量
  await db.collection('skills').doc(skill.data[0]._id).update({
    data: {
      views: _.inc(1)
    }
  });
  
  return {
    code: 200,
    data: skill.data[0]
  };
}

// 搜索技能
async function searchSkills(data) {
  const { query, page = 1, size = 20 } = data;
  
  const skip = (page - 1) * size;
  
  // 使用正则表达式搜索
  const skills = await db.collection('skills')
    .where(_.or([
      { name: db.RegExp({ regexp: query, options: 'i' }) },
      { description: db.RegExp({ regexp: query, options: 'i' }) },
      { tags: query }
    ]))
    .where({ status: 'approved' })
    .skip(skip)
    .limit(size)
    .get();
  
  return {
    code: 200,
    data: skills.data
  };
}

// 创建技能（开发者）
async function createSkill(data) {
  const { userId, skillData } = data;
  
  // 检查 skillId 是否已存在
  const exist = await db.collection('skills')
    .where({ skillId: skillData.skillId })
    .get();
  
  if (exist.data.length > 0) {
    return { code: 400, message: '技能ID已存在' };
  }
  
  const result = await db.collection('skills').add({
    data: {
      ...skillData,
      authorId: userId,
      status: 'pending', // 待审核
      downloads: 0,
      rating: 5.0,
      ratingCount: 0,
      createdAt: db.serverDate(),
      updatedAt: db.serverDate()
    }
  });
  
  return {
    code: 200,
    message: '技能创建成功，等待审核',
    data: { skillId: result._id }
  };
}

// 获取热门技能
async function getPopularSkills(data) {
  const { limit = 10 } = data;
  
  const skills = await db.collection('skills')
    .where({ status: 'approved' })
    .orderBy('downloads', 'desc')
    .limit(limit)
    .get();
  
  return {
    code: 200,
    data: skills.data
  };
}

// 获取新技能
async function getNewSkills(data) {
  const { limit = 10 } = data;
  
  const skills = await db.collection('skills')
    .where({ status: 'approved' })
    .orderBy('createdAt', 'desc')
    .limit(limit)
    .get();
  
  return {
    code: 200,
    data: skills.data
  };
}
```

### 3.3 数据库设计

#### 初始化脚本 (database/init.js)

```javascript
const cloud = require('wx-server-sdk');
cloud.init();

const db = cloud.database();

// 初始化数据库集合和索引
async function initDatabase() {
  // 创建集合
  const collections = ['users', 'skills', 'orders', 'memberships', 'userSkills', 'reviews'];
  
  for (const name of collections) {
    try {
      await db.createCollection(name);
      console.log(`集合 ${name} 创建成功`);
    } catch (e) {
      console.log(`集合 ${name} 已存在`);
    }
  }
  
  // 创建索引
  // 用户集合索引
  await db.collection('users').createIndex({
    data: {
      username: 1
    },
    unique: true
  });
  
  await db.collection('users').createIndex({
    data: {
      email: 1
    },
    unique: true
  });
  
  // 技能集合索引
  await db.collection('skills').createIndex({
    data: {
      skillId: 1
    },
    unique: true
  });
  
  await db.collection('skills').createIndex({
    data: {
      category: 1,
      status: 1
    }
  });
  
  await db.collection('skills').createIndex({
    data: {
      status: 1,
      downloads: -1
    }
  });
  
  console.log('数据库初始化完成');
}

initDatabase();
```

#### Next.js + 云开发 SDK

```javascript
// utils/tcb.js
import tcb from 'tcb-js-sdk';

const app = tcb.init({
  env: 'your-env-id'
});

export const auth = app.auth();
export const db = app.database();
export const storage = app.storage();
export const callFunction = app.callFunction;

// API 封装
export const api = {
  // 用户相关
  async register(data) {
    return callFunction({
      name: 'user-api',
      data: { action: 'register', data }
    });
  },
  
  async login(data) {
    return callFunction({
      name: 'user-api',
      data: { action: 'login', data }
    });
  },
  
  // 技能相关
  async getSkills(params) {
    return callFunction({
      name: 'skill-api',
      data: { action: 'getSkills', data: params }
    });
  },
  
  async getSkillDetail(skillId) {
    return callFunction({
      name: 'skill-api',
      data: { action: 'getSkillDetail', data: { skillId } }
    });
  },
  
  async searchSkills(query) {
    return callFunction({
      name: 'skill-api',
      data: { action: 'searchSkills', data: { query } }
    });
  }
};
```

```javascript
// pages/skills/index.js
import { useState, useEffect } from 'react';
import { api } from '../../utils/tcb';

export default function SkillsPage() {
  const [skills, setSkills] = useState([]);
  const [loading, setLoading] = useState(true);

    loadSkills();

      }
    } catch (error) {
      console.error('加载技能失败:', error);

      setLoading(false);

        {skills.map(skill => (
          <SkillCard key={skill._id} skill={skill} />
        ))}
      </div>
    </div>
  );
}
```

| 对比项 | 传统服务器 | 云开发 |
|--------|-----------|--------|
| 运维成本 | 高（需专人） | 低（免运维） |
| 开发速度 | 慢（需配置环境） | 快（开箱即用） |
| 扩展性 | 需手动配置 | 自动扩缩容 |
| 成本 | ¥2000+/月 | ¥300/月 |
| 学习成本 | 高 | 低 |

### 6.2 适合场景

✅ **适合**：
- MVP 快速验证
- 中小规模应用
- 团队技术资源有限
- 预算有限

❌ **不适合**：
- 超大规模应用（日活100万+）
- 复杂的企业级需求
- 需要高度定制化基础设施

## 七、迁移路径

当业务增长到一定规模时，可以平滑迁移到传统架构：

```
云开发架构 → 混合架构 → 传统架构

阶段1: 云开发（日活 < 10万）
阶段2: 混合（云函数 + 自建数据库，日活 10-50万）
阶段3: 全自建（日活 > 50万）
```
